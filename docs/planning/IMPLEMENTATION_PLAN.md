## Step 2: Create CMakeSection Interface

**New File**: `CMakeGeneration/ICMakeSection.cs`

Create an interface that all CMake sections will implement.

```csharp
using forge.Models;

namespace forge.CMakeGeneration
{
    /// <summary>
    /// Interface for CMake generation sections.
    /// Implement this to add new CMake generation capabilities.
    /// </summary>
    public interface ICMakeSection
    {
        /// <summary>
        /// Unique identifier for this section
        /// </summary>
        string Name { get; }
        
        /// <summary>
        /// Priority determines generation order (lower = earlier)
        /// </summary>
        int Priority { get; }
        
        /// <summary>
        /// Whether this section should be generated
        /// </summary>
        bool IsEnabled(ProjectConfig config);
        
        /// <summary>
        /// Generate CMake content
        /// </summary>
        string Generate(ProjectConfig config);
    }
}
```

## Step 4: Create CMakeRegistry

**New File**: `CMakeGeneration/CMakeRegistry.cs`

Central registry that manages all sections and generates the final CMake.

```csharp
using System.Text;
using forge.Models;

namespace forge.CMakeGeneration
{
    /// <summary>
    /// Registry for CMake sections. Manages registration and generation.
    /// </summary>
    public class CMakeRegistry
    {
        private static CMakeRegistry? _instance;
        public static CMakeRegistry Instance => _instance ??= new CMakeRegistry();
        
        private readonly Dictionary<string, ICMakeSection> _sections = new();
        private bool _initialized = false;
        
        private CMakeRegistry() { }
        
        /// <summary>
        /// Register a section
        /// </summary>
        public void Register(ICMakeSection section)
        {
            _sections[section.Name] = section;
        }
        
        /// <summary>
        /// Initialize with default (built-in) sections
        /// </summary>
        public void Initialize()
        {
            if (_initialized) return;
            
            // Register built-in sections in priority order
            Register(new StandardSection());
            Register(new FetchContentSection());
            Register(new ConanSection());
            Register(new ProjectTargetSection());
            Register(new LinkingSection());
            Register(new CustomCMakeSection());
            Register(new TestingSection());
            
            _initialized = true;
        }
        
        /// <summary>
        /// Generate all enabled sections in priority order
        /// </summary>
        public string Generate(ProjectConfig config)
        {
            Initialize();
            
            var sb = new StringBuilder();
            sb.AppendLine("# This file is generated by Forge. DO NOT TOUCH!");
            sb.AppendLine();
            
            foreach (var section in _sections.Values
                .Where(s => s.IsEnabled(config))
                .OrderBy(s => s.Priority))
            {
                var content = section.Generate(config);
                if (!string.IsNullOrWhiteSpace(content))
                {
                    sb.AppendLine(content);
                    sb.AppendLine();
                }
            }
            
            return sb.ToString();
        }
    }
}
```

```csharp
namespace forge.CMakeGeneration
{
    /// <summary>
    /// Standard C++ configuration (CMAKE_CXX_STANDARD)
    /// </summary>
    public class StandardSection : CMakeSectionBase
    {
        public override string Name => "standard";
        public override int Priority => 1;
        
        private readonly string _standard;
        
        public StandardSection(string standard = "20")
        {
            _standard = standard;
        }
        
        public override string Generate(ProjectConfig config)
        {
            return $@"
set(CMAKE_CXX_STANDARD {_standard})
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)";
        }
    }
}
```

### 5.2 FetchContentSection.cs

```csharp
using forge.Models;
using Spectre.Console;

namespace forge.CMakeGeneration
{
    /// <summary>
    /// Git-based dependencies via FetchContent
    /// </summary>
    public class FetchContentSection : CMakeSectionBase
    {
        public override string Name => "fetchcontent";
        public override int Priority => 10;
        
        public override bool IsEnabled(ProjectConfig config)
            => config.Dependencies.Any();
        
        public override string Generate(ProjectConfig config)
        {
            var sb = new StringBuilder();
            
            sb.AppendLine("include(FetchContent)");
            sb.AppendLine("");
            sb.AppendLine("# --- Dependencies ---");
            
            foreach (var dep in config.Dependencies)
            {
                var name = dep.Key;
                var details = dep.Value;
                
                if (string.IsNullOrEmpty(details.Git) || string.IsNullOrEmpty(details.Tag))
                {
                    AnsiConsole.MarkupLine($"[yellow]Warning:[/] Skipping invalid dependency '{name}'");
                    continue;
                }
                
                sb.AppendLine($"FetchContent_Declare({name} GIT_REPOSITORY \"{details.Git}\" GIT_TAG \"{details.Tag}\")");
                sb.AppendLine($"FetchContent_MakeAvailable({name})");
            }
            
            return sb.ToString();
        }
    }
}
```

### 5.3 LinkingSection.cs

```csharp
using forge.Models;

namespace forge.CMakeGeneration
{
    /// <summary>
    /// Target linking (from FetchContent + Conan + Custom)
    /// </summary>
    public class LinkingSection : CMakeSectionBase
    {
        public override string Name => "linking";
        public override int Priority => 40;
        
        public override string Generate(ProjectConfig config)
        {
            var linkTargets = new List<string>();
            
            // From package.toml dependencies
            foreach (var dep in config.Dependencies)
            {
                if (dep.Key != "googletest")
                {
                    var target = string.IsNullOrEmpty(dep.Value.Target) ? dep.Key : dep.Value.Target;
                    linkTargets.Add(target);
                }
            }
            
            // From Conan
            linkTargets.AddRange(ProjectBuildManager.LinkDependencies);
            
            if (!linkTargets.Any())
                return string.Empty;
            
            return $@"
# --- Linking ---
target_link_libraries({config.Project.Name} PRIVATE {string.Join(" ", linkTargets)})";
        }
    }
}
```

### 5.4 ProjectTargetSection.cs

```csharp
using forge.Models;

namespace forge.CMakeGeneration
{
    /// <summary>
    /// Project target definition (executable or library)
    /// </summary>
    public class ProjectTargetSection : CMakeSectionBase
    {
        public override string Name => "project_target";
        public override int Priority => 30;
        
        public override string Generate(ProjectConfig config)
        {
            var sb = new StringBuilder();
            
            sb.AppendLine("# --- Project Target ---");
            sb.AppendLine($"file(GLOB_RECURSE SOURCES ${{PROJECT_SOURCE_DIR}}/src/*.cpp)");
            
            if (config.Project.Type == "executable")
            {
                sb.AppendLine($"add_executable({config.Project.Name} ${{SOURCES}})");
            }
            else if (config.Project.Type == "library")
            {
                sb.AppendLine($"add_library({config.Project.Name} STATIC ${{SOURCES}})");
                
                if (config.Project.InstallHeaders)
                {
                    sb.AppendLine($"target_include_directories({config.Project.Name} PUBLIC ${{PROJECT_SOURCE_DIR}}/src)");
                    sb.AppendLine($"install(TARGETS {config.Project.Name} EXPORT {config.Project.Name}Config DESTINATION lib)");
                    sb.AppendLine($"install(DIRECTORY ${{PROJECT_SOURCE_DIR}}/src/ DESTINATION include/{config.Project.Name})");
                }
            }
            
            return sb.ToString();
        }
    }
}
```

### 5.5 TestingSection.cs

```csharp
using forge.Models;

namespace forge.CMakeGeneration
{
    /// <summary>
    /// Google Test integration
    /// </summary>
    public class TestingSection : CMakeSectionBase
    {
        public override string Name => "testing";
        public override int Priority => 50;
        
        public override bool IsEnabled(ProjectConfig config)
            => Directory.Exists("test") && config.Dependencies.ContainsKey("googletest");
        
        public override string Generate(ProjectConfig config)
        {
            return @"
# --- Testing ---
enable_testing()

file(GLOB_RECURSE TEST_SOURCES ""${PROJECT_SOURCE_DIR}/test/*.cpp"")

set(SOURCES_FOR_TESTS ${SOURCES})
if(EXISTS ""${PROJECT_SOURCE_DIR}/src/main.cpp"")
    list(REMOVE_ITEM SOURCES_FOR_TESTS ""${PROJECT_SOURCE_DIR}/src/main.cpp"")
endif()

add_executable(run_tests ${TEST_SOURCES} ${SOURCES_FOR_TESTS})
target_include_directories(run_tests PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src)
target_link_libraries(run_tests PUBLIC GTest::gtest_main)

include(GoogleTest)
gtest_discover_tests(run_tests)";
        }
    }
}
```

## Step 6: Create Custom CMake Section

**New File**: `CMakeGeneration/Sections/CustomCMakeSection.cs`

This is the key section that reads from Lua scripts.

```csharp
using forge.Models;

namespace forge.CMakeGeneration
{
    /// <summary>
    /// Custom CMake from Lua build scripts
    /// </summary>
    public class CustomCMakeSection : CMakeSectionBase
    {
        public override string Name => "custom";
        public override int Priority => 45; // After linking, before testing
        
        public override bool IsEnabled(ProjectConfig config)
            => ProjectBuildManager.CustomCMakeSnippets.Any();
        
        public override string Generate(ProjectConfig config)
        {
            var sb = new StringBuilder();
            sb.AppendLine("# --- Custom CMake (from build scripts) ---");
            
            foreach (var snippet in ProjectBuildManager.CustomCMakeSnippets)
            {
                // Replace ${PROJECT_NAME} placeholder with actual name
                var resolved = snippet.Replace("${PROJECT_NAME}", config.Project.Name);
                sb.AppendLine(resolved);
            }
            
            return sb.ToString();
        }
    }
}
```

## Step 7: Add Lua Function for Custom CMake

**File**: `Commands/Lua/LuaEngine.cs`

Add a function that allows Lua scripts to append custom CMake.

```csharp
private static void SetCustomCMakeFunction()
{
    var addCmakeFunc = new LuaFunction("add_cmake", (context, token) =>
    {
        var cmakeSnippet = context.GetArgument<string>(0);
        
        // Add to ProjectBuildManager for CMakeGenerator to use
        ProjectBuildManager.CustomCMakeSnippets.Add(cmakeSnippet);
        
        AnsiConsole.MarkupLine($"[green]Added custom CMake snippet[/]");
        
        return ValueTask.FromResult(0);
    });
    
    _cpm[new LuaValue("add_cmake")] = new LuaValue(addCmakeFunc);
}
```

Then call it in `SetDefinitionTables()`:

```csharp
private static void SetDefinitionTables(ref LuaState state)
{
    SetGitFunctions();
    SetLoggingFunctionsAndDefinitions();
    SetEnvironmentVariableInformation();
    SetGetPackagesFunction();
    SetCustomCMakeFunction();  // Add this line
    
    state.Environment["forge"] = _cpm;
}
```
