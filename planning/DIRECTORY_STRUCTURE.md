## Generated CMake Structure

The generated `.config/cmake/CMakeLists.txt` will look like:

```cmake
# This file is generated by Forge. DO NOT TOUCH!

# --- Standard ---
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# --- Dependencies (Conan) ---
find_package(fmt REQUIRED)

# --- Dependencies ---
include(FetchContent)
FetchContent_Declare(googletest GIT_REPOSITORY "..." GIT_TAG "...")
FetchContent_MakeAvailable(googletest)

# --- Project Target ---
file(GLOB_RECURSE SOURCES ${PROJECT_SOURCE_DIR}/src/*.cpp)
add_executable(myproject ${SOURCES})

# --- Linking ---
target_link_libraries(myproject PRIVATE SDL2::SDL2)

# --- Custom CMake (from build scripts) ---
# Hello from Lua!

# --- Testing ---
enable_testing()
# ... test configuration
```
